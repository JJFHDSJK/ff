        function initializeMap() {
            map = L.map('map').setView([31.5, -7.5], 6);

            satellite.addTo(map);

            // Gestion des changements de fond
            document.getElementById('satellite-layer').addEventListener('change', function() {
                if (this.checked) {
                    map.removeLayer(streets);
                    map.removeLayer(terrain);
                    satellite.addTo(map);
                    currentBasemap = 'satellite';
                }
            });

            document.getElementById('street-layer').addEventListener('change', function() {
                if (this.checked) {
                    map.removeLayer(satellite);
                    map.removeLayer(terrain);
                    streets.addTo(map);
                    currentBasemap = 'streets';
                }
            });

            document.getElementById('terrain-layer').addEventListener('change', function() {
                if (this.checked) {
                    map.removeLayer(satellite);
                    map.removeLayer(streets);
                    terrain.addTo(map);

                }
            });

            // Charger les symboles des r√©gions
            loadRegionSymbols();
        }

        // Fonction pour obtenir la couleur selon la valeur
        function getColorForValue(value, variable) {
            const scale = colorScales[variable];
            for (let range of scale) {
                if (value >= range.min && value < range.max) {
                    return range.color;
                }
            }
            return '#3498db';
        }

        // Fonction pour obtenir l'ic√¥ne selon la variable
        function getIconForVariable(value, variable, isSelected = false) {
            const color = isSelected ? '#e74c3c' : getColorForValue(value, variable);
            const size = isSelected ? 25 : 20;
            
            // Diff√©rents symboles selon la variable
            let symbol;
            switch(variable) {
                case 'temperature':
                    symbol = 'fa-thermometer-half';
                    break;
                case 'precipitation':
                    symbol = 'fa-cloud-rain';
                    break;
                case 'wind':
                    symbol = 'fa-wind';
                    break;
                default:
                    symbol = 'fa-map-marker-alt';
            }
            
            return L.divIcon({
                className: 'region-symbol' + (isSelected ? ' selected' : ''),
                html: `<div style="
                    background-color: ${color}; 
                    width: ${size}px; 
                    height: ${size}px; 
                    border-radius: 50%; 
                    border: 3px solid white;
                    display: flex;
                    align-items: center;
                    justify-content: center;
                    box-shadow: 0 2px 5px rgba(0,0,0,0.3);
                    font-size: ${size * 0.6}px;
                    color: white;
                "><i class="fas ${symbol}"></i></div>`,
                iconSize: [size, size],
                iconAnchor: [size/2, size/2]
            });
        }

        // Cr√©er les symboles pour les r√©gions
        function loadRegionSymbols() {
            Object.entries(regionsData2024).forEach(([regionName, regionData]) => {
                const value = regionData.monthlyData[currentVariable][currentMonth];
                const isSelected = regionName === currentSelectedRegion;
                
                // Cr√©er un marqueur avec symbole
                const marker = L.marker(regionData.coords, {
                    icon: getIconForVariable(value, currentVariable, isSelected)
                }).addTo(map);

                // Stocker la r√©f√©rence
                regionLayers[regionName] = marker;

                // Info-bulle
                const unit = getUnitForVariable(currentVariable);
                marker.bindTooltip(`
                    <div style="text-align: center; color: #333; min-width: 150px;">
                        <strong>${regionName}</strong><br>
                        ${getVariableLabel(currentVariable)}: ${value}${unit}
                    </div>
                `);

                // Gestion des clics
                marker.on('click', function() {
                    // R√©initialiser le style pr√©c√©dent
                    if (currentSelectedRegion && regionLayers[currentSelectedRegion]) {
                        const prevValue = regionsData2024[currentSelectedRegion].monthlyData[currentVariable][currentMonth];
                        regionLayers[currentSelectedRegion].setIcon(
                            getIconForVariable(prevValue, currentVariable, false)
                        );
                    }

                    // Mettre en surbrillance la r√©gion s√©lectionn√©e
                    marker.setIcon(getIconForVariable(value, currentVariable, true));

                    currentSelectedRegion = regionName;

                    // Mettre √† jour l'interface
                    document.getElementById("region-name").innerHTML = regionName;
                    updateRegionInfo(regionName, regionData);
                    updateChart(regionName);
                });
            });
        }

        // Mettre √† jour les statistiques
        function updateStatistics() {
            const regions = Object.keys(regionsData2024);
            
            let totalTemp = 0;
            let regionCount = 0;
            
            const hottest = regions.reduce((a, b) => 
                regionsData2024[a].monthlyData.temperature[currentMonth] > 
                regionsData2024[b].monthlyData.temperature[currentMonth] ? a : b
            );
            const wettest = regions.reduce((a, b) => 
                regionsData2024[a].monthlyData.precipitation[currentMonth] > 
                regionsData2024[b].monthlyData.precipitation[currentMonth] ? a : b
            );
            const windiest = regions.reduce((a, b) => 
                regionsData2024[a].monthlyData.wind[currentMonth] > 
                regionsData2024[b].monthlyData.wind[currentMonth] ? a : b
            );
            
            regions.forEach(region => {
                totalTemp += regionsData2024[region].monthlyData.temperature[currentMonth];
                regionCount++;
            });
            
            document.getElementById('hottest-region').textContent = hottest;
            document.getElementById('wettest-region').textContent = wettest;
            document.getElementById('windiest-region').textContent = windiest;
            document.getElementById('national-avg').textContent = (totalTemp / regionCount).toFixed(1) + "¬∞C";
        }
